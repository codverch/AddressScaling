#!/bin/bash

# Install GCC 10
sudo add-apt-repository ppa:ubuntu-toolchain-r/test -y
sudo apt-get update
sudo apt-get install gcc-10 g++-10 -y
sudo update-alternatives --install /usr/bin/gcc gcc /usr/bin/gcc-10 20
sudo update-alternatives --install /usr/bin/g++ g++ /usr/bin/g++-10 20
sudo update-alternatives --config gcc
sudo update-alternatives --config g++

# Clone the LLVM project repository
git clone https://github.com/llvm/llvm-project.git

# Navigate to the LLVM project directory
cd llvm-project

# Configure and build LLVM with AddressSanitizer support
mkdir build
cd build
cmake -DLLVM_ENABLE_PROJECTS="clang" -DLLVM_USE_SANITIZER=Address -DCMAKE_BUILD_TYPE=Release -DCMAKE_C_COMPILER=gcc-10 -DCMAKE_CXX_COMPILER=g++-10 -G Ninja ../llvm
ninja

# Run LLVM regression tests
ninja check-all

# Optionally, install LLVM
ninja install
